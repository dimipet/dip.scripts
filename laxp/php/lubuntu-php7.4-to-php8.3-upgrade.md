# disable php7.4 and enable php8.3
```
$ sudo a2dismod php7.4
$ apache2ctl -M
$ sudo systemctl restart apache2
$ sudo systemctl status apache2
$ sudo a2enmod php8.3
$ sudo systemctl restart apache2
$ sudo systemctl status apache2
$ apache2ctl -M
```
